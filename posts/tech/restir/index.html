Class Reservoir{
	y = 0;
	W_sum = 0;
	M = 0;
	W = 0;
	update(x,w_i){
		w_sum = w_sum + w_i;
		M = M + 1;
		if(rand() < (w_i/w_sum)) y = x_i
	}
}

reservoirSampling(Samples){
	Reservoir r;
	for i = 1 to M
		r.update(Samples[i], weight[Samples[i]])
	return r
}
